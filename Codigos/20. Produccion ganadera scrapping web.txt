const municipio = document.getElementById("municipio");

function esperar(ms) {
	return new Promise(resolve => setTimeout(resolve, ms));
}

async function ejecutarConRetardo() {
	for (let opciones of municipio.options) {
		document.getElementById("municipio").value = opciones.value;
		console.log(opciones.value);
		reporte();
		descargarExcel();

		// Esperar 5 segundos
		await esperar(5000);
	}
}

ejecutarConRetardo();
